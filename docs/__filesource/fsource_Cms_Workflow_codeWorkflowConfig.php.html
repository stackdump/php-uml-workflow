  80.             {
  81.                 $id   = $state['xmi.id'];
  82.                 $name = $state['name'];
  83.  
  84.                 $node_meta = array( 
  85.                         'id'            => $id, 
  86.                         'name'          => $name, 
  87.                         'isFirstNode'   => $isFirstNode,
  88.                         'isLastNode'    => $isLastNode

  90.  
  91.                 // call the factory method to build a node
  92.                 $node = Cms_Workflow_Factory::Node($node_meta);
  93.  
  94.                 /**

  96.                  *  in and out transitions are added with different methods
  97.                  */
  98.                 $transition_types = array('addInTransition' => 'UML:StateVertex.incoming', 'addOutTransition' => 'UML:StateVertex.outgoing');
  99.  
  100.                 /**
  101.                  *  add both in and out transistions
  102.                  */
  103.                 foreach($transition_types as $addTransFunc => $type)
  104.                 {
  105.                     /**
  106.                      *  Add transitions to all node types
  107.                      */
  108.                     if( isset($state[$type]) && is_array($state[$type]))
  109.                     {
  110.                         /**
  111.                          *  cycle through all state types 
  112.                          */
  113.                         foreach ($state[$type] as $transitions)
  114.                         {
  115.                             /**
  116.                              * Check to see if this is a single record or an array
  117.                              */
  118.                             if( isset($transitions['UML:Transition']['xmi.idref']))
  119.                             {
  120.                                $transition_ids = array( 0 => $transitions['UML:Transition']);
  121.                             }
  122.                             else // more than one transition
  123.                             {
  124.                                $transition_ids = $transitions['UML:Transition'];
  125.                             }
  126.                             
  127.                             /**
  128.                              *  create a transition object for every id
  129.                              */
  130.                             foreach($transition_ids as $row)
  131.                             {
  132.                                 $transition_meta = $this->lookupTransition($row['xmi.idref']);
  133.                                 $transition_meta['type'] = $type;
  134.  
  135.                                 //echo "<pre>";  print_r($transition_meta); echo "</pre>";
  136.  
  137.                                 /**
  138.                                  *  Setup transitions then add condition and trigger/eventSignal 
  139.                                  */
  140.                                 if(! empty($transition_meta))
  141.                                 {
  142.                                     $TransitionObj = Cms_Workflow_Factory::Transition($transition_meta);
  143.                                     $condition_meta = $this->lookupCondition($row['xmi.idref']);
  144.                                     $TransitionObj->setCondition( Cms_Workflow_Factory::Condition($condition_meta ));
  145.  
  146.                                     if(!$transition_meta['trigger'])
  147.                                     {
  148.                                         //echo "<pre>";  print_r($state); echo "</pre>";
  149.                                         throw new Cms_Workflow_Config_Exception('A Trigger is required for every transition.');
  150.                                     }
  151.  
  152.                                     $signal_meta = array('name' => $this->lookupTriggerEvent($transition_meta['trigger']));
  153.                                     $TransitionObj->setWorkflowSignal(Cms_Workflow_Factory::Signal($signal_meta));
  154.  
  155.  
  156.                                     /**

  159.                                     $effect_meta =  $this->lookupEffect($row['xmi.idref']);
  160.                                     if(is_array($effect_meta))
  161.                                     {
  162.                                         //echo "<pre>";  var_dump(array($row , $effect_meta)); echo "</pre>";
  163.                                         $TransitionObj->setEffect(Cms_Workflow_Factory::Effect($effect_meta));
  164.                                     }
  165.  
  166.                                     //finally add the transition to the Node
  167.                                     $node->$addTransFunc($TransitionObj);
  168.                                 }
  169.                             }
  170.                         }
  171.                     }
  172.                 }
  173.  
  174.                 /**
  175.                  *  Set actions for this state
  176.                  */
  177.                 if(! ($isFirstNode || $isLastNode))
  178.                 {
  179.                     
  180.                      //  allow mapping of types to the appropriate set function
  181.                      //  todo allow other types of actions
  182.                     $actionTypesFunc = array(
  183.                         'UML:State.doActivity' => 'setDoActivity', 
  184.                         'UML:State.entry' => 'setEntry');
  185.  
  186.                     $actions= self::getActionsMeta($state);
  187.  
  188.                     //echo '<pre>'; print_r($state); echo '</pre>'; die;
  189.                     //echo '<pre>'; print_r($actions); echo '</pre>'; die;
  190.                     foreach($actions as $type => $action_meta)
  191.                     {
  192.                         //echo '<pre>'; print_r($action_meta); echo '</pre>'; die;
  193.                         $setFunc = $actionTypesFunc[$type];
  194.                         $node->$setFunc(Cms_Workflow_Factory::Action($action_meta));
  195.                     }
  196.                 }
  197.  
  198.                 $nodes[$id] = $node;
  199.             }
  200.         }
  201.  
  202.         return $nodes;

  417.     /**
  418.      * getTansitionId - extract state id from transitions
  419.      *

  421.      *
  422.      *  EX: 'UML:SignalEvent' =>
  423.      *    array
  424.      *         'xmi.idref' => string '127-0-1-1--2166a281:12522aaa6e2:-8000:0000000000001113' (length=54)

  427.      * @access private

  429.      */
  430.     private function _extractStateId($state_array)
  431.     {
  432.         if(! is_array($state_array)) return;
  433.
